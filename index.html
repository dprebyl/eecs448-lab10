EECS 448 Lab 10 (MySQL)

Exercise 1 SQL provided below. Exercises 2-7 accessible via the navbar above. 
CREATE TABLE Users ( 
	user_id varchar(255) NOT NULL, 
	PRIMARY KEY (user_id) 
);

CREATE TABLE Posts ( 
	post_id int NOT NULL AUTO_INCREMENT, 
	content text, 
	author_id varchar(255) NOT NULL, 
	PRIMARY KEY (post_id), 
	FOREIGN KEY (author_id) REFERENCES Users(user_id) 
);
